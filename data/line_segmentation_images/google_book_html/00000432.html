འཛམ་གླིང་ཚོང་དོན་ཚོགས་པའི་སྐོར་གྱི་གཞི་རྩའི་ཤེས་བྱ།

བས་མཐོ་བའི་ཞབས་ཞུ་ཉོ་ཚོང་རང་དབང་ཅན་དུ་མངོན་འགྱུར་བྱེད་པའི་ཆེད་ དུ། ཁོངས་ཞུགས་ཕྱོགས་ཀྱིས་ག་ཚོད་འཕྱི་ནའང་2000ལོའི་ཟླ་1པོར་རེས་ སྐོར་གྲོས་ མོལ་ གསར་ པ་ བྱ་ དགོས་ ཞེས་ གཏན་ འབེབས་བྱས་ཡོད། = འབྱུང་ འགྱུར་གྱི་གྲོས་མོལ་བྱེད་པའི་སློབ་སྟོན་རྩ་དོན་གཏན་ཁེལ་བྱེད་སྐབས།

ཞབས་

ཞུ་ཉོ་ ཚོང་ ལས་ འཛིན་ ལྷན་ ཚོགས་ ཀྱིས་འཕེལ་ རྒྱས་ འགྲོ་ བཞིན་ པའི་ ཁོངས་ ཞུགས་ཕྱོགས་ཀྱིས་དོ་ཁུར་ཆེས་ཆེར་བྱེད་པའི་གནད་དོན་གཉིས་ཏེ་གཅིག་ནི་ ཁོངས་ཞུགས་ཕྱོགས་ཁག་གིས་རེས་སྐོར་གྲོས་མོལ་གོང་མ་བྱས་པ་ནས་བཟུང་གི་ ཚོང་རའི་སྒོ་མོ་འབྱེད་པའི་འབད་བརྩོན་བྱས་ཚུལ་ལ་ལྟ་སྟངས་ཇི་ལྟར་འཛིན་ དགོས་པ་དང་། ཅིག་ཤོས་ནི་དར་རྒྱས་ཆུང་ཤོས་ཡིན་པའི་ཁོངས་ཞུགས་ཕྱོགས་ ལ་དམིགས་བསལ་གྱི་ཐོབ་ཐང་སྤྲོད་དགོས་པའི་ཐད་ལ་ཆོད་དོན་འཇོག་དགོས་

པ་ཡིན།

བཞི། སྒྲིག་གཞིའི་ཐད་ཀྱི་དོན་ཚན་དང་མཐའ་མའི་དོན་ཚན་སྐོར། 《ཞབས་ཞུ་ཉོ་ཚོང་སྐོར་གྱི་སྤྱིའི་ཆོད་དོན་》ནང་འཁོད་སྒྲིག་གཞིའི་དོན་ ཚན་དང་མཐའ་མའི་དོན་ཚན་དེ་དག་ནི་“ཝུ་ར་ཀུའེ་རེས་སྐོར་གྲོས་མོལ་”" གྱི་ གྲོས་མཐུན་གཞན་དག་ནང་གསལ་སྒྲིག་གཞིའི་དོན་ཚན་དང་མཐའ་མའི་དོན་ ཚན་དང་ འདྲ་ མཚུངས་ཡིན། སྒྲིག་གཞིའི་དོན་ཚན་ནང་ ལ་ རྩོད་ གཞི་ ཐག་ གཅོད་བྱེད་སྟངས་དང་། ཞབས་ཞུ་ཉོ་ཚོང་ལས་འཛིན་ལྷན་ཚོགས་འཛུགས་ སྟངས། ལག་རྩལ་གྱི་མཉམ་ལས་བྱེད་སྟངས། རྒྱལ་སྤྱིའུ་རྩ་འཛུགས་གཞན་ དག་བར་དང་འབྲེལ་བ་བྱེད་སྟངས་སོགས་ཀྱི་ཐད་གཏན་འབེབས་བྱས་ཡོད། མཐའ་མའི་ དོན་ ཚན་ ནང་ ཁོངས་ ཞུགས་ ཕྱོགས་ ཀྱིས་ ཞབས་ ཞུ་ ཉོ་ ཚོང་ དང་ འབྲེལ་ཡོད་ ཀྱི་ ཁ་ ཆད་ བཞག་དོན་ ཞིབ་ ཕྲ་ དེ་ཁོངས་ ཞུགས་ ཕྱོགས་ མིན་ པའི་ ཞབས་ཞུའམ་ཞབས་ཞུ་མཁོ་སྤྲོད་བྱེད་མཁན་ལ་མ་སྤྲད་ཀྱང་ཆོག་ཅེས་འཁོད་ ཡོད། མཐའ་མའི་དོན་ཚན་ནང་《ཞབས་ཞུ་ཉོ་ཚོང་སྐོར་གྱི་སྤྱིའི་ཆོད་དོན་》

ནང་བེད་སྤྱོད་བྱས་པའི་ཐ་སྙད་གཙོ་བོ་འགའ་ཤས་དཔེར་ན་“ ཞབས་ཞུ་མཁོ་
