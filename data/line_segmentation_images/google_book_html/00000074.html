འཛམ་གླིང་ཚོང་དོན་ཚོགས་པའི་སྐོར་གྱི་གཞི་རྩའི་ཤེས་བྱ།

འགོག་གམ་ཚོད་འཛིན་སོགས་ཀྱི་བྱེད་ཐབས། ཕྱིར་གཏོང་ནང་འདྲེན་བྱེད་ རིགས་དངུལ་ཁོངས་ ག་ བསྒྱུར་གཏོང་བར་སྤྱོད་པའི་བཀག་འགོག་གམ་ཚོད་ འཛིན་བྱེད་ཐབས། ཕྱིར་གཏོང་ནང་འདྲེན་གྱི་ཐོན་རྫས་བྲིན་ཚོང་དང་། བགོ་ ཚོང་། སྐྱེལ་འདྲེན། འགན་བཅོལ། མཛོད་འཇུག ཞིབ་བཤེར། འགྲེམས་ སྟོན། ཞིབ་ལས་བཅས་བྱེད་པར་ཤུགས་རྐྱེན་ཐེབས་རིགས་དང་རྒྱལ་ནང་གི་ ཐོན་རྫས་དང་ མཉམ་ དུ་ སྤྱོད་པར་གནོད་པའམ་ཡང་ ན་ སྤྱོད་ མཁོ་ གཞན་ ལ་ གཏོང་བར་གནོད་པའི་བླང་བྱ། ཞབས་ཞུ་ཉོ་ཚོང་དང་ འབྲེལ་བ་ཡོད་པའི་ བཅའ་ཁྲིམས་དང་། ཁྲིམས་སྲོལ། སྲིད་ཇུས། བྱེད་ཐབས་བཅས་ཡིན། ཤེས་ ཡོན་གྱི་ཐོན་དངོས་བདག་དབང་དང་འབྲེལ་བ་ཡོད་པའི་བཅའ་ཁྲིམས་དང་། ཁྲིམས་སྲོལ། ཁྲིམས་འཛིན་གྱི་དཔྱད་མཚམས་དང་སྲིད་འཛིན་གྱི་ཁྲུན་ཐག་ བཅད་དོན་ བཅས་ དང་ དེ་ བཞིན་ འཛམ་ གླིང་ ཚོང་ དོན་ ཚོགས་ པའི་ ཁོངས་ ཞུགས་ཕྱོགས་ཀྱིས་མིང་རྟགས་བཀོད་ཅིང་རྒྱལ་སྤྱིའི་ཉོ་ཚོང་སྲིད་ཇུས་ལ་ཤུགས་

རྐྱེན་ཐེབས་པའི་གྲོས་མཐུན་གཞན་དག་སོགས་ཡིན།

ཡོངས་བསྒྲགས་བྱ་ རྒྱུའི་ དུས་ཚོད་སྐོར་ལ་ཁོངས་ཞུགས་ཕྱོགས་ཀྱིས་ཉོ་ ཚོང་དང་ འབྲེལ་ བ་ ཡོད་པའི་བཅའ་ཁྲིམས་དང༌། ཁྲིམས་སྲོལ། སྲིད་ཇུས། བྱེད་ཐབས། ཁྲིམས་འཛིན་གྱི་དཔྱད་མཚམས། སྲིད་འཛིན་ གྱི་ ཁྲུན་ ཐག་ བཅད་དོན་ བཅས་ འཕྲལ་ མར་ ཡོངས་ བསྒྲགས་ དང་ གསལ་ ཚད་ ལྡན་ པ་ བྱ་ དགོས་ཤིང་། ག་ཚོད་འཕྱི་ནའང་དེ་དག་གོ་ཆོད་པར་གྱུར་དུས་ནས་ཡོངས་

བསྒྲགས་སམ་ གསལ་ ཚད་ ལྡན་ པ་ བྱས་ ཏེ་ འཛམ་ གླིང་ ཚོང་ དོན་ ཚོགས་ པའི་ ཁོངས་ཞུགས་ཕྱོགས་གཞན་དག་དང་ཉོ་ཚོང་བྱེད་མཁན་ཚོང་པ་རྣམས་ཀྱིས་ དུས་ཐོག་ཏུ་ཤེས་ཐུབ་པ་བྱ་དགོས་ཞེས་འཛམ་གླིང་ཚོང་དོན་ཚོགས་པས་གཏན་ འབེབས་བྱས་ཡོད། ཡོངས་བསྒྲགས་མ་བྱས་གོང་ལ་བྱེད་ཐབས་སྔོན་དུ་སྤྱད་མི་ ཆོག་ཅིང་། ནང་འདྲེན་ཐོན་རྫས་ཀྱི་སྒོ་ཁྲལ་གྱི་ཁྲལ་ཚད་དམ་འགྲོ་གྲོན་གཞན་

དག་མཐོ་རུ་གཏོང་བ་དང་། ནང་འདྲེན་ཐོན་རྫས་སམ་ནང་འདྲེན་ཐོན་རྫས་
