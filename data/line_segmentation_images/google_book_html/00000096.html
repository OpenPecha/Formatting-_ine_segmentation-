འཛམ་གླིང་ཚོང་དོན་ཚོགས་པའི་སྐོར་གྱི་གཞི་རྩའི་ཤེས་བྱ།

བསྟར་བའི་འགྲན་རྩོད་དྲང་གཞག་བྱ་རྒྱུའི་རྩ་དོན།

ཤེས་བྱའི་ཐོན་དངོས་བདག་དབང་ཁྱབ་ཁོངས་ནང་ལག་ལེན་བསྟར་བའི་ འགྲན་རྩོད་དྲང་གཞག་བྱ་རྒྱུའི་རྩ་དོན་ནི་གཙོ་བོར་ཤེས་བྱའི་ཐོན་དངོས་བདག་ དབང་སྲུང་སྐྱོང་ནུས་ལྡན་བྱ་རྒྱུ་དང་ཚུལ་མིན་གྱི་འགྲན་རྩོད་ལ་ངོ་རྒོལ་བྱ་རྒྱུར་

མཚོན་ཡོད།

《ཉོ་ཚོང་དང་འབྲེལ་ཡོད་ཀྱི་ཤེས་བྱའི་ཐོན་དངོས་བདག་དབང་སྐོར་གྱི་ ཆོད་དོན་》གྱི་ནང་དུ་ཁོངས་ཞུགས་ཕྱོགས་སོ་སོས་ཤེས་བྱའི་ཐོན་དངོས་བདག་ དབང་སྲུང་ སྐྱོང་ བྱེད་ ཤུགས་ ཆེ་ རུ་ བཏང་ ནས་ ཤེས་ བྱའི་ ཐོན་ དངོས་ བདག་ དབང་ལྡན་པའི་ཐོན་རྫས་དང་ཚོང་རྟགས་འདྲ་བཟོ་དང༌། རྫུན་སྒྲིག རྐུ་སྤྱོད་ བཅས་བྱེད་ཚུལ་སྔོན་འགོག་བྱ་དགོས་པའི་བླང་བྱ་བཏོན་ཡོད། ཁབ་ཀྱི་རྒྱལ་དམངས་ཀྱི་ཤེས་བྱའི་ཐོན་དངོས་བདག་དབང་ཡིན་ནའང་འདྲ། ཡང་ན་ ཁོངས་ ཞུགས་ ཕྱོགས་ གཞན་ དག་ གི་ རྒྱལ་ དམངས་ ཀྱི་ ཤེས་ བྱའི་ ཐོན་ དངོས་བདག་དབང་ཡིན་ནའང་འདྲ་ཚང་མར་སྲུང་སྐྱོང་ནུས་ལྡན་བྱ་དགོས། ཆོད་དོན་འདིའི་ནང་དུ་འགྲན་རྩོད་དྲང་གཞག་བྱེད་པའི་ཚོང་རའི་ཁོར་

རང་གི་རྒྱལ་

ཡུག་ཅིག་གསར་སྐྲུན་བྱེད་ཆེད་ཁོངས་ཞུགས་ཕྱོགས་སོ་སོས་ གཟིགས་སྐྱོང་ཆེ་ ཤོས་བྱེད་ སའི་ རྒྱལ་ཁབ་ཀྱི་ཐོབ་ ཐང་ སྤྲོད་ རྒྱུའི་ལག་ལེན་བསྟར་ནས་ཁོངས་ ཞུགས་ཕྱོགས་མི་འདྲ་བ་གཞན་དག་ནས་ཡོང་བའི་རྒྱལ་དམངས་ཀྱི་ཤེས་ཡོན་གྱི་ ཐོན་དངོས་བདག་དབང་ལ་སྲུང་སྐྱོང་འདྲ་མཚུངས་འཐོབ་པ་བྱ་དགོས་པ་དང་། དུས་མཚུངས་ སུ་ རྒྱལ་ དམངས་ཀྱི་ ཐོབ་ ཐང་ སྤྲོད་ རྒྱུའི་ ལག་ལེན་ བསྟར་ ནས་ ཁོངས་ཞུགས་ཕྱོགས་གཞན་དག་ནས་ཡོང་བའི་ རྒྱལ་ དམངས་ ཤེས་བྱའི་ཐོན་ དངོས་བདག་དབང་ལ་ རང་གི་ རྒྱལ་ཁབ་ཀྱི་རྒྱལ་དམངས་ཀྱི་ ཤེས་བྱའི་ ཐོན་ དངོས་བདག་དབང་དང་འདྲ་མཚུངས་ཡིན་པའི་སྲུང་སྐྱོང་འཐོབ་པ་བྱ་དགོས་

པའི་བླང་བྱའང་བཏོན་ཡོད

ཚུལ་མིན་ གྱི་ འགྲན་ རྩོད་ ལ་ངོ་རྒོལ་བྱ་ རྒྱུ་དེའང་ ཤེས་བྱའི་ཐོན་དངོས་
