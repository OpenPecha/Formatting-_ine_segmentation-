གཉིས། སྒོ་ཁྲལ་ཉུང་འཕྲི་བྱས་ཚུལ།

སྒོ་ཁྲལ་གསལ་ཚད་ ཆེ་བ་ཡིན་ན། བསྡུར་ཞིབ་བྱེད་བདེ་བ་ཡིན་མོད། འོན་ཀྱང་ཕྱིར་གཏོང་ནང་འདྲེན་བྱ་རྒྱུའི་ཚོང་ཟོག་གི་རིན་གོང་ལ་ཐད་ཀར་

ཤུགས་རྐྱེན་ཐེབས་རིགས་སྤྱི་དང༌། ལྷག་པར་དུ་སྒོ་ཁྲལ་མཐོ་རིགས་དེ་ནི་རྒྱལ་ སྤྱིའི་ཐོག་ དངོས་ ཟོག་ རང་ དགར་ འགྲོ་ རྒྱུག་ བྱེད་ པར་ ཚོད་ འཛིན་ བྱེད་ པའི་ འགལ་རྐྱེན་གཙོ་བོ་ཞིག་རེད། དེར་བརྟེན་ཉོ་ཚོང་རང་མོས་ཅན་དུ་འགྱུར་བར་

སྐུལ་འདེད་རྒྱུན་ཆད་མེད་པར་གཏོང་ཆེད་དུ་འཛམ་གླིང་ཚོང་དོན་ཚོགས་པས་ ཁོངས་ཞུགས་ཕྱོགས་སོ་སོར་སྒོ་ཁྲལ་གྱི་བྱེད་ཐབས་བེད་སྤྱོད་བྱེད་དུ་འཇུག་པ་

དང་དུས་ མཚུངས་ སུ་ ཁོངས་ ཞུགས་ ཕྱོགས་ རྣམས་ ལ་ སྒོ་ ཁྲལ་ སྡུད་ ཚད་ རིམ་ བཞིན་དམའ་རུ་གཏོང་བ་དང་འབྲེལ་ཚོད་འཛིན་ཡང་བྱ་དགོས་པའི་བླང་བྱ་ བཏོན་ཡོད། “ སྒོ་ཁྲལ་ཚོད་འཛིན་བྱེད་པ” ཞེས་པ་ནི་ཁོངས་ཞུགས་ཕྱོགས་ ཀྱིས་ནང་འདྲེན་བྱ་རྒྱུའི་ཚོང་ཟོག་གི་སྒོ་ཁྲལ་སྡུད་ཚད་དེ་ཚད་ག་གེ་མོ་ཞིག་གི་ ཐོག་ ཚོད་ འཛིན་ བྱས་ ནས་ མཐོ་ རུ་ མི་ གཏོང་ རྒྱུའི་ ཁ་ དན་ བཞག་ པར་ ཟེར། དཔེར་ན་ཁོངས་ཞུགས་ཕྱོགས་གཅིག་གིས་དཀའ་ངལ་དངོས་ཡོད་པའི་དབང་ གིས་སྒོ་ཁྲལ་ཚོད་འཛིན་བྱེད་ཚད་དེ་མཐོ་རུ་གཏོང་དགོས་ཚེ་ངེས་པར་དུ་ཁོངས་ ཞུགས་ཕྱོགས་གཞན་དག་དང་ཡང་བསྐྱར་གྲོས་མོལ་བྱ་དགོས་པ་ལྟ་བུ་རེད། ཕྱོགས་མང་ཉོ་ཚོང་གྲོས་མོལ་སྔ་མ་ཐེངས་བདུན་བྱས་པའི་རྨང་གཞིའི་

སྟེང་“ ཝུ་ ར་ཀུའེ་རེས་སྐོར་གྲོས་མོལ་” ཐོག་ཁོངས་ཞུགས་ཕྱོགས་སོ་སོའི་ཐོན་
